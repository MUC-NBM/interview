    # server.js

    var http = require("http");

http.createServer(function(request, response){
  response.writeHead(200,{"Content-Type":"text/plain"});
  response.write("Hello World");
  response.end();

    接下来，打开浏览器访问http://localhost:8888/，你会看到一个写着“Hello World”的网页。

    第一行请求（require）Node.js自带的 http 模块，并且把它赋值TypeScript 类、泛型的使用实践记录给 http 变量。 接下来我们调用http模块提供的函数： createServer 。这个函数会返回一个对象，这个对象有一个叫做 listen 的方法，这个方法有一个数值参数，指定这个HTTP服务器监听的端口号。

    var http = require("http");

    # 进行函数传递

    传递函数的时候不必写()

    function say(word){
  console.log(word);
}

function execute(someFunction, value){
  someFunction(value);
}

execute(say,"Hello");

    可以不用先定义再传递，定义和传递可以同时在execute中执行，即通过匿名函数实现

    function execute(someFunction, value){
  someFunction(value);
}

execute(function(word){ console.log(word)},"Hello");

    关于回调函数的理解： 我们创建了服务器，并且向创建它的方法传递了一个函数，无论何时我们的服务器收到一个请求(就是访问浏览器地址)，服务器就会调用我们发过去的函数。这个过程就叫做回调。

    # 解析服务器的驱动过程

    var http = require("http");

function onRequest(request, response){
  console.log("Request received.");
  response.writeHead(200,{"Content-Type":"text/plain"});
  response.write("Hello World");
  response.end();
}

http.createServer(onRequest).listen(8888);

console.log("Server has started.");

    通过shell运行node demo.js之后会运行这串js代码，然后会输出这行代码console.log("Server has started.");``,但是当我们访问http://localhost:8888/ `才会运行函数，“Request received.”这条消息就会在命令行中出现。

    # 服务器是如何处理请求的

    当回调启动，我们的 onRequest() 函数被触发的时候，有两个参数被传入： request 和 response 。

    它们是对象，你可以使用它们的方法来处理HTTP请求的细节，并且响应请求（比如向发出请求的浏览器发回一些东西）。

    所以我们的代码就是：当收到请求时，使用 response.writeHead() 函数发送一个HTTP状态200和HTTP头的内容类型（content-type），使用 response.write() 函数在HTTP相应主体中发送文本“Hello World"。

    最后，我们调用 response.end() 完成响应。

    # 如何来进行请求的路由

    我们需要查看HTTP请求，从中提取出请求的URL以及GET/POST参数，随后路由需要根据这些数据来执行相应的代码。 我的疑问

    在一个HTTP请求中，request.url是一个包含了完整URL的字符串，它是客户端（例如浏览器）在访问网站时发送给服务器的请求中的一部分。 当你访问 http://www.example.com/products/shoes 这个网址时，浏览器会向服务器发送一个HTTP请求，其中包含了请求的URL。服务器接收到这个请求后，会将请求的URL存储在 request.url 中，以便你的代码可以访问和处理这个URL。 所以，在你的代码中，url.parse(request.url) 将会解析 request.url 这个字符串，并返回一个URL对象，其中包含了URL的各个部分，如协议、主机名、路径名等。通过 .pathname 属性，我们可以从URL对象中提取出路径名。

    # 阻塞与非阻塞

    阻塞与非阻塞的异步（Asynchronous）是一种编程模式，它指的是在执行某个操作时，程序可以继续执行后续的代码，而不必等待当前操作完成。换句话说，异步操作允许程序在等待某些任务的同时执行其他任务，而不会被阻塞。

    在传统的同步编程模式中，当执行一个耗时的操作时（如文件读写、网络请求等），程序会一直等待该操作完成后才继续往下执行。这可能会导致程序的执行流程被阻塞，而在这段时间内，程序无法执行其他任务。定义：

    同步=阻塞是指在执行一个操作时，程序会一直等待该操作完成后才会继续执行下面的代码。 异步=非阻塞是指在执行一个操作时，程序会立即继续执行后续代码，而不会等待该操作的完成。在非阻塞模式下，程序会通过回调函数、事件监听或者Promise等方式，在操作完成后得到通知，并处理操作的结果。

    nodejs是单线程的的，通过事件轮询来实现并行操作，也就是说，任何一个进程的阻塞操作都会影响后续进程，所以在nodejs的程序中应该尽量避免程序中有阻塞程序。 理解nodejs的事件轮询

    # 阻塞举例

    function start(){
  console.log("Request handler 'start' was called.");

  function sleep(milliSeconds){
    var startTime =new Date().getTime();
    while(new Date().getTime()< startTime + milliSeconds);
  }

  sleep(10000);
  return"Hello Start";
}

function upload(){
  console.log("Request handler 'upload' was called.");
  return"Hello Upload";
}

exports.start = start;
exports.upload = upload;

    上述代码中，当函数start()被调用的时候，Node.js会先等待10秒，之后才会返回“Hello Start”。当调用upload()的时候，会和此前一样立即返回。 但是，假如你先打开/start，然后马上打开/upload，/upload URL居然也花了10秒，而它在对应的请求处理程序中并没有类似于sleep()这样的操作！

    这是因为sleep()是阻塞的，因此浏览器会等到/start中的sleep()执行完毕才会执行upload。所以如果你先打开upload就不会发生阻塞。（nodejs为单线程） 因此Node.js可以在不新增额外线程的情况下，依然可以对任务进行并行处理 —— Node.js是单线程的。它通过事件轮询（event loop）来实现并行操作，对此，我们应该要充分利用这一点 —— 尽可能的避免阻塞操作，取而代之，多使用非阻塞操作。

    # 非阻塞举例

    var exec = require("child_process").exec;

function start(){
  console.log("Request handler 'start' was called.");
  var content ="empty";

  exec("ls -lah",function(error, stdout, stderr){
    content = stdout;
  });

  return content;
}

function upload(){
  console.log("Request handler 'upload' was called.");
  return"Hello Upload";
}

exports.start = start;
exports.upload = upload;
    1. 当exec()被调用之后启动"ls-lah" 子进程并立即返回。
    2. 主线程继续执行后续代码，此时 content 的值还是 "empty"。也就是说主进程会执行return content。
    3. "ls -lah" 子进程执行完毕，回调函数被调用，将命令输出赋值给 content。
    4. 由于回调函数的异步性质，这时 content 才被更新为正确的值。但是此时return早已完成。

    # 阻塞与非阻塞

    如果想要证明/start处理程序中耗时的操作不会阻塞对/upload请求作出立即响应的话，可以将requestHandlers.js修改为如下形式：

    var exec = require("child_process").exec;

function start(response){
  console.log("Request handler 'start' was called.");

  exec("find /",
    { timeout:10000, maxBuffer:20000*1024},
    function(error, stdout, stderr){
      response.writeHead(200,{"Content-Type":"text/plain"});
      response.write(stdout);
      response.end();
    });
}

function upload(response){
  console.log("Request handler 'upload' was called.");
  response.writeHead(200,{"Content-Type":"text/plain"});
  response.write("Hello Upload");
  response.end();
}

exports.start = start;
exports.upload = upload;

    这样一来，当请求http://localhost:8888/start的时候，会花10秒钟的时间才载入，而当请求http://localhost:8888/upload的时候，会立即响应，纵然这个时候/start响应还在处理中。

    { timeout:10000, maxBuffer:200001024},与sleep()并没有区别，只是因为 { timeout:10000, maxBuffer:200001024},在exec()中才不会阻塞

    # 非阻塞函数再探讨

    Node.js 就是一个非阻塞式的运行时环境，它广泛使用了异步编程模型，允许程序能够同时处理多个操作而不阻塞主线程。这使得 Node.js 可以高效地处理并发请求，从而提供高性能的网络应用程序。

    当涉及 Node.js 时，异步非阻塞编程模型是核心概念。以下是一些示例，展示了如何在 Node.js 中使用异步非阻塞函数来处理操作：

    1. 文件读取：
    const fs = require('fs');

// 异步读取文件
fs.readFile('file.txt', 'utf8', (err, data) => {
  if (err) {
    console.error('Error:', err);
    return;
  }
  console.log('File content:', data);
});

console.log('Reading file...');  // 不会等待文件读取完成，会立即执行
    1. 网络请求：
    const http = require('http');

// 异步发起 HTTP 请求
http.get('http://example.com', (response) => {
  let data = '';

  response.on('data', (chunk) => {
    data += chunk;
  });

  response.on('end', () => {
    console.log('Received data:', data);
  });
});

console.log('Making HTTP request...');  // 不会等待请求响应，会立即执行
    1. 定时器：
    // 使用 setTimeout 实现异步定时器
console.log('Start');
setTimeout(() => {
  console.log('Delayed function');
}, 2000);  // 两秒后执行，不会阻塞主线程
console.log('End');
    1. 回调函数：
    // 使用回调函数实现异步操作
function fetchData(callback) {
  setTimeout(() => {
    callback('Data fetched');
  }, 1000);
}

console.log('Start');
fetchData((data) => {
  console.log(data);  // 在 fetchData 完成后执行
});
console.log('End');

    这些示例展示了在 Node.js 中如何使用异步非阻塞操作。通过利用回调、事件处理、Promise、async/await 等机制，Node.js 能够在执行耗时操作时保持高效和高响应性，而不会阻塞主线程。这使得 Node.js 成为构建高性能、实时和并发应用程序的强大工具。 <!--
